$nerine.affection = 0
$nerine.diaryUnlocked = false
$nerine.minigame.Lune.completed = false
$nerine.minigame.Lune.perfect = false
$nerine.diary.Lune.unlocked = false

CreateCharacter(Nerine -i true -e true)

setLayerMedia(background spider_lily)

Nerine "..."
Nerine "This place... feels familiar."

choice "What do you feel?"
{
    -Warm
        $nerine.affection += 5
    -Confused
        $nerine.affection += 3
    -Nothing
        $nerine.affection += 1
}

Nerine "Fragments... I can't see them clearly."

if ($nerine.affection >= 5)
{
    narrator "A memory begins to surface..."
    playminigame(Lune -char Nerine -scene Minigame_Memory1)
}

Nerine "..."

if ($nerine.minigame.Lune.completed == true)
{
    if ($nerine.minigame.Lune.perfect == true)
    {
        Nerine "You remembered everything... perfectly."
    }
    else
    {
        Nerine "Even if it's incomplete... thank you."
    }
}

if ($nerine.diary.Lune.unlocked == true)
{
    narrator "New diary entry unlocked: Lune"
}
if ($nerine.diaryUnlocked == true)
{
    narrator "Diary panel is now accessible."
}

Nerine "Maybe... we can remember more together."
